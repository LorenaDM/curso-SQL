-- length
Select length("Tiago");
select titulo, length(titulo) as "Tamanho do Título"
 from livro where id=1;

-- lower
select lower(titulo) from livro where id=1;

-- upper()

-- substr(x,y,z)

-- random()
select random();

from livro;

-- The replace(X,Y,Z)

-- round(x)
